# Quick Start: Node.js Example

📗 **TUTORIAL**: A complete, runnable Node.js program that performs the entire quick start flow: connect to the IOTC broker, discover the reader, start an RFID inventory, print tag data for 10 seconds, and stop.

## Prerequisites

- Node.js 16 or later
- `mqtt` package (mqtt.js)

Initialize a project and install the dependency:

```bash
npm init -y
npm install mqtt
```

## Complete Program

```javascript
/**
 * Zebra IoT Connector; Quick Start Example (Node.js)
 * Connects to the IOTC MQTT broker, discovers the reader,
 * starts a simple inventory, prints tags for 10 seconds, and stops.
 */

const mqtt = require("mqtt");

// ──────────────────────────────────────────────
// Configuration; replace with your actual values
// ──────────────────────────────────────────────
const BROKER_HOST = "mqtt.iotconnector.zebra.com";
const BROKER_PORT = 8883;
const MQTT_USERNAME = "tenant-user@abc123def456";
const MQTT_PASSWORD = "your-mqtt-password";
const TENANT_ID = "abc123def456";
const DEVICE_SERIAL = "RFD40-1234567890";

// ──────────────────────────────────────────────
// Topic construction
// ──────────────────────────────────────────────
const MGMT_CMD = `${TENANT_ID}/MGMT/clients/cmnd/${DEVICE_SERIAL}`;
const MGMT_RESP = `${TENANT_ID}/MGMT/clients/resp/${DEVICE_SERIAL}`;
const CTRL_CMD = `${TENANT_ID}/CTRL/clients/cmnd/${DEVICE_SERIAL}`;
const CTRL_RESP = `${TENANT_ID}/CTRL/clients/resp/${DEVICE_SERIAL}`;
const DATA_EVT = `${TENANT_ID}/DATA/clients/data1event/${DEVICE_SERIAL}`;

let tagCount = 0;

function publishCommand(client, topic, command, requestId, payload) {
  const msg = { command, requestId };
  if (payload) msg.payload = payload;
  client.publish(topic, JSON.stringify(msg));
  console.log(`[publish] ${command} (requestId: ${requestId})`);
}

function sleep(ms) {
  return new Promise((resolve) => setTimeout(resolve, ms));
}

async function main() {
  const brokerUrl = `mqtts://${BROKER_HOST}:${BROKER_PORT}`;

  console.log(`Connecting to ${BROKER_HOST}:${BROKER_PORT}...`);

  const client = mqtt.connect(brokerUrl, {
    username: MQTT_USERNAME,
    password: MQTT_PASSWORD,
    protocolVersion: 4, // MQTT 3.1.1
    rejectUnauthorized: true,
  });

  client.on("connect", () => {
    console.log("[connected] Successfully connected to IOTC broker");
    client.subscribe([MGMT_RESP, CTRL_RESP, DATA_EVT], (err) => {
      if (err) {
        console.error("[error] Subscription failed:", err.message);
      } else {
        console.log("[subscribed] Listening on MGMT, CTRL, and DATA topics");
      }
    });
  });

  client.on("error", (err) => {
    console.error("[error] Connection error:", err.message);
  });

  client.on("message", (topic, message) => {
    let payload;
    try {
      payload = JSON.parse(message.toString());
    } catch {
      console.log(`[raw] ${topic}: ${message.toString()}`);
      return;
    }

    if (topic === MGMT_RESP) {
      const cmd = payload.command || "unknown";
      const status = payload.status || "unknown";
      console.log(`[mgmt] ${cmd}: ${status}`);
      if (cmd === "get_version" && status === "OK") {
        const info = payload.payload || {};
        console.log(`  Model: ${info.model}`);
        console.log(`  Serial: ${info.serialNumber}`);
        console.log(`  Firmware: ${info.firmwareVersion}`);
      }
    } else if (topic === CTRL_RESP) {
      const cmd = payload.command || "unknown";
      const status = payload.status || "unknown";
      const state = (payload.payload || {}).operationState || "";
      console.log(`[ctrl] ${cmd}: ${status} (state: ${state})`);
    } else if (topic === DATA_EVT) {
      const tags = (payload.payload || {}).tags || [];
      for (const tag of tags) {
        tagCount++;
        console.log(
          `[tag #${tagCount}] EPC: ${tag.epc}  RSSI: ${tag.rssi} dBm`
        );
      }
    }
  });

  // Wait for connection
  await sleep(3000);

  // Step 1: Discover reader
  publishCommand(client, MGMT_CMD, "get_version", "1");
  await sleep(2000);

  // Step 2: Set operating mode
  publishCommand(client, CTRL_CMD, "set_operating_mode", "2", {
    mode: "SIMPLE_INVENTORY",
  });
  await sleep(2000);

  // Step 3: Start inventory
  publishCommand(client, CTRL_CMD, "control_operation", "3", {
    action: "START",
  });

  // Step 4: Read tags for 10 seconds
  console.log("\n--- Reading tags for 10 seconds ---\n");
  await sleep(10000);

  // Step 5: Stop inventory
  console.log(`\n--- Stopping inventory (${tagCount} tags read) ---\n`);
  publishCommand(client, CTRL_CMD, "control_operation", "4", {
    action: "STOP",
  });
  await sleep(2000);

  // Disconnect
  client.end(false, () => {
    console.log(`[done] Total tags read: ${tagCount}`);
  });
}

main().catch(console.error);
```

## Running the Program

1. Save the code as `iotc_quickstart.js`.
2. Update the configuration variables at the top of the file with your actual IOTC credentials.
3. Run:

```bash
node iotc_quickstart.js
```
