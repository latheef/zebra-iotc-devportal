import BrowserOnly from '@docusaurus/BrowserOnly';
import { usePluginData } from '@docusaurus/useGlobalData';

export function AsyncApiPage() {
  const pluginData = usePluginData('docusaurus-plugin-asyncapi', 'plugin-asyncapi-3');
  const schema = pluginData?.asyncapiSpec;
  if (!schema) return <p>Loading...</p>;
  const AsyncApiComponent = require('@asyncapi/react-component/browser');
  return (
    <AsyncApiComponent
      schema={schema}
      config={{ show: { sidebar: true } }}
    />
  );
}

<BrowserOnly fallback={<p>Loading...</p>}>
  {() => <AsyncApiPage />}
</BrowserOnly>
